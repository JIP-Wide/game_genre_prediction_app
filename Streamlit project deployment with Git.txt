Wenn du das Projekt lokal mit Git verwaltest, lädst du Ordner und Dateien ganz einfach hoch:
git init
git add .
git commit -m "Initial commit"
git remote add origin https://github.com/DEIN_BENUTZERNAME/DEIN_REPO.git
git branch -M main
git push -u origin main

git lfs install
git lfs track *.csv
git push -u origin main

conda deactivate
conda remove --name dein_env_name --all

conda create -n neuer_env_name python=3.10  # oder deine gewünschte Version
conda activate neuer_env_name
pip install -r requirements.txt

git add requirements.txt  # oder alle Änderungen: git add .
git commit -m "Update dependencies and recreate clean environment"
git push origin main
